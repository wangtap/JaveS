登录：
	mysql -uroot -p123
创建数据库：
	create database 库名;
创建表
	create table 表名(
		列名1 类型1,
		列名2 类型2,
		列名3 类型3
	);
添加数据
	insert into 表名 values(
		第一列的值,
		第二列的值,
		第三列的值
	);
查找数据
	select * from 表名;	
	*表示的是要把所有的列都查找出来
更新数据
	update 表名 set 列1=值1;
	把列1中所有的值都设置为值1
删除数据
	delete from 表名;
	把表中的数据全删了。

删除表
	drop table 表名;
删除库
	drop database 库名;

查找表的基本结构
	desc 表名;

进入某个数据库
	use 库名;
查看某个数据库中所有的表
	show tables;
查看mysql中所有的库
	show databases;

退出
	quit;

如果语句无法结束：
	打个分号后回车
如果分号也不管用：
	看看是不是少了一个单引号
	所以打个：';
	回车
遇到带near的错误的时候：
	就是语句打错了，找不到错误就从打

show variables like 'character%';
